import { PageHeaderDemo } from '@/components/demos/composed-demo';

`PageHeader` renders the top of a page: a required `title` heading, an optional
`description`, a `breadcrumbs` trail above the title, and an `actions` slot
pinned to the right for buttons. It handles the spacing and responsive wrapping
so every screen's header looks consistent. Pair it with
[`PageBody`](/docs/components/pagebody) for the content below.

## Example

<Demo>
  <PageHeaderDemo />
</Demo>

```tsx
import { PageHeader, Button } from '@officina/ui';

<PageHeader
  title="Invoices"
  description="Manage and export billing documents"
  breadcrumbs={[{ label: 'Home', href: '/' }, { label: 'Invoices' }]}
  actions={<Button>New invoice</Button>}
/>;
```
