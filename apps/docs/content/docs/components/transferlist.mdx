---
title: TransferList
description: Dual-pane picker for moving items between an available and a selected list.
---

import { TransferListDemo } from '@/components/demos/data-display-demo';

`TransferList` shows two panes — available on the left, selected on the right —
and lets users check items and move them between panes with the arrow buttons.
It is controlled: `value` holds the ids of selected items and `onChange` fires
with the next selection. Rename the panes with `availableLabel` and
`selectedLabel`. Use it for assigning permissions, members, or tags.

## Example

<Demo>
  <TransferListDemo />
</Demo>

```tsx
import { TransferList } from '@officina/ui';
import { useState } from 'react';

function Permissions() {
  const [value, setValue] = useState<string[]>(['read']);
  return (
    <TransferList
      items={[
        { value: 'read', label: 'Read' },
        { value: 'write', label: 'Write' },
        { value: 'delete', label: 'Delete' },
      ]}
      value={value}
      onChange={setValue}
    />
  );
}
```
