import { SearchInputDemo } from '@/components/demos/inputs-demo';

`SearchInput` is a controlled text field with a leading search icon and a clear
(×) button that appears once there is a value. Beyond the immediate `onChange`,
it offers `onDebouncedChange` that fires after `debounceMs` (default 200ms) of
inactivity — wire your actual query there to avoid firing a request on every
keystroke. Disable the clear button with `showClear={false}`.

## Example

<Demo>
  <SearchInputDemo />
</Demo>

```tsx
import { SearchInput } from '@officina/ui';
import { useState } from 'react';

function Example() {
  const [query, setQuery] = useState('');
  return (
    <SearchInput
      value={query}
      onChange={setQuery}
      onDebouncedChange={(q) => runSearch(q)}
      placeholder="Search…"
    />
  );
}
```
