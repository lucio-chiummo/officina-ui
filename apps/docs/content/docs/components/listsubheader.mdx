import { ListSubheaderDemo } from '@/components/demos/composed-demo';

`ListSubheader` is a non-interactive `<li>` that renders a small, uppercase,
muted caption to title a section of a `List`. Use it to group related rows —
"Navigation", "Account", "Danger zone" — so longer lists stay scannable. Pass the
label as children.

## Example

<Demo>
  <ListSubheaderDemo />
</Demo>

```tsx
import { List, ListItem, ListItemButton, ListSubheader } from '@officina/ui';

<List>
  <ListSubheader>Navigation</ListSubheader>
  <ListItem>
    <ListItemButton>Dashboard</ListItemButton>
  </ListItem>
  <ListItem>
    <ListItemButton>Settings</ListItemButton>
  </ListItem>
</List>;
```
