---
title: DateTimeRangePicker
description: Picks a start and end date-time range, each with its own day and time of day.
---

import { DateTimeRangePickerDemo } from '@/components/demos/inputs-demo';

`DateTimeRangePicker` captures a start and end date-time as a `DateRange`
(`{ from, to }`), where each end carries both a date and a time of day. It is
controlled via `value`/`onChange`. Use it for bookings, shifts, or any window
that needs precise start/end times; label the two ends with `startLabel` and
`endLabel`.

## Example

<Demo>
  <DateTimeRangePickerDemo />
</Demo>

```tsx
import { DateTimeRangePicker, type DateRange } from '@officina/ui';
import { useState } from 'react';

function BookingWindow() {
  const [range, setRange] = useState<DateRange>({ from: undefined, to: undefined });
  return (
    <DateTimeRangePicker
      value={range}
      onChange={setRange}
      startLabel="Check in"
      endLabel="Check out"
    />
  );
}
```
