import { TabsDemo } from '@/components/demos/tabs-demo';

`Tabs` switches between panels from a declarative `items` array.

## Example

<Demo>
  <TabsDemo />
</Demo>

```tsx
import { Tabs } from '@officina/ui';

<Tabs
  items={[
    { label: 'Overview', content: <Overview /> },
    { label: 'Usage', content: <Usage /> },
  ]}
/>;
```
