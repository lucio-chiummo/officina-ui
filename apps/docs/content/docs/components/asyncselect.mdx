import { AsyncSelectDemo } from '@/components/demos/composed-demo';

`AsyncSelect` is a [`Combobox`](/docs/components/combobox) that fetches its
options from a server as the user types. You supply `queryFn` (returns a promise
of `ComboboxOption`s for the query) and `queryKey` (the React Query cache key for
that query); loading and empty states are handled for you. Use `minChars` to
avoid firing requests on short input and `minCharsMessage` to prompt the user.
It is controlled via `value`/`onChange` like a regular Combobox.

## Example

<Demo>
  <AsyncSelectDemo />
</Demo>

```tsx
import { AsyncSelect } from '@officina/ui';
import { useState } from 'react';

function UserSearch() {
  const [value, setValue] = useState<string | null>(null);
  return (
    <AsyncSelect
      value={value}
      onChange={setValue}
      minChars={2}
      placeholder="Search users…"
      queryKey={(query) => ['users', query]}
      queryFn={async (query) => {
        const res = await fetch(`/api/users?q=${query}`);
        const users = await res.json();
        return users.map((u) => ({ value: u.id, label: u.name }));
      }}
    />
  );
}
```
