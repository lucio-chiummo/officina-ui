import { CardFooterDemo } from '@/components/demos/composed-demo';

`CardFooter` is the bottom section of a `Card`. It draws a top border and lays
its children out in a row, making it the place for action buttons or summary
metadata. Place it as the last child of a `Card`, after the content.

## Example

<Demo>
  <CardFooterDemo />
</Demo>

```tsx
import { Card, CardContent, CardFooter, Button } from '@officina/ui';

<Card>
  <CardContent>Growth plan — renews May 11, 2026.</CardContent>
  <CardFooter>
    <Button size="sm" variant="soft">
      Manage
    </Button>
  </CardFooter>
</Card>;
```
